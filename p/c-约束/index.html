template<typename T>
concept Component = requires(T t)
{
    t.step1();
    {t.step2()} -> std::convertible_to<bool>;
    t.step3();
    {t.step4()} -> std::integral;
    t.step5();
};

class Library
{
public:
    static void process(Component auto& component)
    {
        component.step1();
        if (component.step2())
        {
            component.step3();
        }
        if (component.step4() < 10)
        {
            component.step5();
        }
    }

    void step1()
    {
        cout << format("Library::step1\n");
    }

    bool step2()
    {
        cout << format("Library::step2\n");
        return true;
    }

    void step3()
    {
        cout << format("Library::step3\n");
    }
};

class App : public Library
{
public:
   int step4()
   {
       cout << format("App::step4\n");
       return 5;
   }

   void step5()
   {
       cout << format("App::step5\n");
   }
};
